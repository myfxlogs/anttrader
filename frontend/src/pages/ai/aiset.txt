import { useMemo } from 'react'
import {
  Bot,
  Key,
  RefreshCw,
  Check,
  AlertCircle,
  ExternalLink,
  Sparkles,
  Zap,
  ShieldCheck,
  Link2,
  Cpu,
  Eraser,
} from 'lucide-react'
import {
  Input,
  Button,
  Select,
  Alert,
  Space,
  Switch,
  Slider,
  Checkbox,
  InputNumber,
  Tooltip,
  Tag,
  Empty,
} from 'antd'

import { ALL_PURPOSES, PROVIDER_LINKS } from './systemai/constants'
import { useSystemAIPage } from './systemai/hooks'

// ---------- 厂商展示元数据（仅用于 UI 呈现，不影响后端协议）----------
type ProviderMeta = {
  label: string
  tagline: string
  icon: React.ComponentType<{ className?: string }>
  accent: string // tailwind 背景色
}

const PROVIDER_META: Record<string, ProviderMeta> = {
  openai: { label: 'OpenAI', tagline: 'GPT 系列 · 官方', icon: Sparkles, accent: 'bg-emerald-50 text-emerald-700 border-emerald-200' },
  anthropic: { label: 'Anthropic', tagline: 'Claude 系列', icon: ShieldCheck, accent: 'bg-amber-50 text-amber-700 border-amber-200' },
  deepseek: { label: 'DeepSeek', tagline: '深度求索 · 高性价比', icon: Zap, accent: 'bg-blue-50 text-blue-700 border-blue-200' },
  moonshot: { label: 'Moonshot', tagline: 'Kimi · 长上下文', icon: Cpu, accent: 'bg-indigo-50 text-indigo-700 border-indigo-200' },
  qwen: { label: '通义千问', tagline: '阿里云 · 中文优化', icon: Sparkles, accent: 'bg-purple-50 text-purple-700 border-purple-200' },
  zhipu: { label: '智谱 GLM', tagline: '清华系 · 通用', icon: Bot, accent: 'bg-cyan-50 text-cyan-700 border-cyan-200' },
  openai_compatible: { label: '自定义（OpenAI 兼容）', tagline: '任意兼容端点', icon: Link2, accent: 'bg-slate-50 text-slate-700 border-slate-200' },
}

function metaOf(providerId: string, fallbackName: string): ProviderMeta {
  return (
    PROVIDER_META[providerId] || {
      label: fallbackName || providerId,
      tagline: '自定义 Provider',
      icon: Bot,
      accent: 'bg-slate-50 text-slate-700 border-slate-200',
    }
  )
}

export default function SystemAI() {
  const {
    configs,
    loading,
    savingConfig,
    savingSecret,
    selectedProviderId,
    setSelectedProviderId,
    draft,
    setDraft,
    secretInput,
    setSecretInput,
    notice,
    error,
    validated,
    setValidated,
    validating,
    discovering,
    setLastAutoDiscoverKey,
    load,
    saveConfig,
    clearSecret,
    validateConnection,
  } = useSystemAIPage()

  const hasSecret = !!(secretInput.trim() || draft?.has_secret)

  // 保留 validateConnection 引用：在自动验证失败时，用户仍可通过刷新触发（隐性兑现）。
  void validateConnection

  const urlDiagnostics = useMemo(() => {
    const value = (draft?.base_url || '').trim()
    if (!value) return { ok: false, https: false }
    try {
      const u = new URL(value)
      return { ok: u.protocol === 'http:' || u.protocol === 'https:', https: u.protocol === 'https:' }
    } catch {
      return { ok: false, https: false }
    }
  }, [draft?.base_url])

  const overallStatus: { tone: 'success' | 'warning' | 'error' | 'info'; title: string; desc: string } = useMemo(() => {
    if (!draft) return { tone: 'info', title: '尚未选择厂商', desc: '请从下方卡片挑选一个模型厂商开始配置' }
    if (error) return { tone: 'error', title: '存在异常', desc: error }
    if (validated && draft.enabled) return { tone: 'success', title: '运行就绪', desc: `${metaOf(draft.provider_id, draft.name).label} · ${draft.default_model || '未选模型'} 已启用并连接正常` }
    if (validated) return { tone: 'warning', title: '连接正常，尚未启用', desc: '打开「启用」开关即可投入使用' }
    if (hasSecret && urlDiagnostics.ok) return { tone: 'info', title: '配置已就绪', desc: '选择默认模型后系统将自动完成连通性检测' }
    if (hasSecret) return { tone: 'warning', title: '请检查 Base URL', desc: 'API Key 已就绪，但地址似乎无效' }
    return { tone: 'info', title: '请完成密钥配置', desc: '填写 API Key 后将自动发现模型列表' }
  }, [draft, error, validated, hasSecret, urlDiagnostics.ok])

  const selectedMeta = draft ? metaOf(draft.provider_id, draft.name) : null

  // ---------- Render ----------
  return (
    <div className="space-y-6 pb-24 max-w-5xl mx-auto">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold text-gray-900 flex items-center gap-2">
            <Bot className="w-6 h-6 text-blue-600" /> 系统 AI 配置
          </h1>
          <p className="text-sm text-gray-500 mt-1">统一管理大模型服务商、API 密钥与默认模型；支持 OpenAI 协议兼容端点。</p>
        </div>
        <Button icon={<RefreshCw className="w-4 h-4" />} onClick={load} loading={loading}>
          刷新
        </Button>
      </div>

      {/* 总体状态横幅 */}
      {!loading && configs.length > 0 && (
        <StatusBanner
          tone={overallStatus.tone}
          title={overallStatus.title}
          description={overallStatus.desc}
          notice={draft && !error ? notice : ''}
        />
      )}

      {/* Loading / Empty */}
      {loading && (
        <div className="text-center py-16 bg-white rounded-xl shadow-sm border border-gray-100">
          <RefreshCw className="w-8 h-8 animate-spin mx-auto text-blue-600" />
          <p className="text-gray-500 mt-3">加载中...</p>
        </div>
      )}

      {!loading && configs.length === 0 && (
        <div className="bg-white rounded-xl shadow-sm border border-gray-100 p-12">
          <Empty description="暂无 AI Provider 配置（系统启动时会自动创建默认 Provider）" />
        </div>
      )}

      {/* ① 厂商选择（卡片网格，首屏即可见，无默认选择） */}
      {!loading && configs.length > 0 && (
        <Section
          step={1}
          title="选择模型厂商"
          subtitle="卡片直接展示每个厂商的配置与就绪状态，点击选择"
        >
          <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-3">
              {configs.map((cfg) => {
                const m = metaOf(cfg.provider_id, cfg.name)
                const Icon = m.icon
                const ready = cfg.has_secret && !!cfg.default_model
                const isSelected = cfg.provider_id === selectedProviderId
                const stateLabel = !cfg.has_secret
                  ? { text: '未配置', color: 'default' as const }
                  : !cfg.default_model
                    ? { text: '待选模型', color: 'warning' as const }
                    : cfg.enabled
                      ? { text: '已启用', color: 'success' as const }
                      : { text: '已就绪 · 未启用', color: 'processing' as const }
                return (
                  <button
                    key={cfg.provider_id}
                    type="button"
                    onClick={() => {
                      if (cfg.provider_id === selectedProviderId) return
                      setSelectedProviderId(cfg.provider_id)
                      setValidated(false)
                    }}
                    className={`text-left rounded-lg border p-3 transition-all hover:shadow-sm hover:border-blue-300 ${
                      isSelected ? 'border-blue-500 ring-2 ring-blue-100 bg-blue-50/40' : 'border-gray-200 bg-white'
                    }`}
                  >
                    <div className="flex items-center gap-3">
                      <div className={`w-9 h-9 rounded-md flex items-center justify-center border shrink-0 ${m.accent}`}>
                        <Icon className="w-4 h-4" />
                      </div>
                      <div className="flex-1 min-w-0">
                        <div className="flex items-center gap-2">
                          <span className="font-medium text-gray-900 truncate">{m.label}</span>
                          {isSelected && <Tag color="blue" style={{ marginInlineEnd: 0 }}>当前</Tag>}
                        </div>
                        <div className="text-xs text-gray-500 truncate">{m.tagline}</div>
                      </div>
                      <Tag color={stateLabel.color} style={{ marginInlineEnd: 0 }}>{stateLabel.text}</Tag>
                    </div>
                    <div className="mt-2 flex items-center gap-1.5 flex-wrap text-xs">
                      <Tag color={cfg.has_secret ? 'blue' : 'default'} style={{ marginInlineEnd: 0 }}>
                        {cfg.has_secret ? '已配密钥' : '未配密钥'}
                      </Tag>
                      <Tag color={cfg.default_model ? 'geekblue' : 'default'} style={{ marginInlineEnd: 0 }}>
                        {cfg.default_model ? `默认模型: ${cfg.default_model}` : '未设默认模型'}
                      </Tag>
                      <Tag style={{ marginInlineEnd: 0 }}>
                        {`模型 ${cfg.models?.length ?? 0} 个`}
                      </Tag>
                      {!ready && cfg.enabled && (
                        <Tag color="error" style={{ marginInlineEnd: 0 }}>启用但不可用</Tag>
                      )}
                    </div>
                  </button>
                )
              })}
            </div>
          </Section>
      )}

      {!loading && draft && selectedMeta && (
        <>
          {/* ② 连接配置 */}
          <Section
            step={2}
            title={`连接配置 · ${selectedMeta.label}`}
            subtitle={
              PROVIDER_LINKS[draft.provider_id] ? (
                <a
                  href={PROVIDER_LINKS[draft.provider_id]}
                  target="_blank"
                  rel="noreferrer"
                  className="text-xs text-blue-600 hover:underline inline-flex items-center gap-1"
                >
                  <ExternalLink className="w-3 h-3" /> 前往申请 / 管理该厂商 API Key
                </a>
              ) : null
            }
          >
            <div className="space-y-4">
              {/* Base URL */}
              <div>
                <Label
                  text="Base URL（模型服务地址）"
                  hint={
                    draft.provider_id === 'openai_compatible'
                      ? '输入 OpenAI 兼容端点，例如 https://model.myfxlogs.org/v1'
                      : '官方地址由系统维护，不可修改'
                  }
                />
                <Input
                  size="large"
                  value={draft.base_url}
                  onChange={(e) => {
                    if (draft.provider_id !== 'openai_compatible') return
                    setDraft({ ...draft, base_url: e.target.value })
                    setValidated(false)
                    setLastAutoDiscoverKey('')
                  }}
                  placeholder={
                    draft.provider_id === 'openai_compatible'
                      ? '例如: https://model.myfxlogs.org/v1'
                      : '官方地址（只读）'
                  }
                  disabled={draft.provider_id !== 'openai_compatible'}
                />
                {draft.base_url && !urlDiagnostics.https && (
                  <p className="text-xs text-amber-700 flex items-center gap-1 mt-1.5">
                    <AlertCircle className="w-3.5 h-3.5" /> 当前为 HTTP，生产环境建议使用 HTTPS
                  </p>
                )}
              </div>

              {/* API Key */}
              <div>
                <Label
                  text="API Key"
                  hint="输入后将自动加密保存，无需手动提交"
                  badge={draft.has_secret ? <Tag color="blue" icon={<Key className="w-3 h-3" />}>已配置</Tag> : undefined}
                />
                <Space.Compact style={{ width: '100%' }}>
                  <Input.Password
                    size="large"
                    value={secretInput}
                    onChange={(e) => setSecretInput(e.target.value)}
                    placeholder={draft.has_secret ? '如需更换密钥，请重新输入' : '粘贴 API Key，将自动预保存'}
                  />
                  <Button
                    size="large"
                    icon={<Eraser className="w-4 h-4" />}
                    onClick={clearSecret}
                    disabled={savingSecret || !draft.has_secret}
                    loading={savingSecret}
                  >
                    删除密钥
                  </Button>
                </Space.Compact>
              </div>

              {/* 默认模型 */}
              <div>
                <Label
                  text="默认模型"
                  hint="保存基础信息后将自动通过 /models 发现可用模型"
                  badge={
                    discovering ? (
                      <span className="text-xs text-gray-500 flex items-center gap-1">
                        <RefreshCw className="w-3 h-3 animate-spin" /> 自动拉取中
                      </span>
                    ) : undefined
                  }
                />
                <Select
                  size="large"
                  value={draft.default_model || undefined}
                  onChange={(v) => {
                    setDraft({ ...draft, default_model: v })
                    setValidated(false)
                  }}
                  options={(draft.models || []).map((m) => ({ value: m, label: m }))}
                  style={{ width: '100%' }}
                  showSearch
                  placeholder="请选择或搜索模型"
                  notFoundContent={<span className="text-xs text-gray-500">暂无模型，请先完成 Base URL 与 API Key</span>}
                />
              </div>
            </div>
          </Section>

          {/* ③ 高级参数（常驻） */}
          <Section
            step={3}
            title="高级参数"
            subtitle="仅在了解含义时调整；默认值已适配大多数场景"
          >
            <div className="space-y-6">
                <div>
                  <Label text="启用状态" hint="关闭后该厂商不参与系统路由" />
                  <label className="flex items-center gap-3 cursor-pointer select-none">
                    <Switch checked={draft.enabled} onChange={(v) => setDraft({ ...draft, enabled: v })} />
                    <span className="text-sm text-gray-700">
                      {draft.enabled ? (
                        <span className="text-green-700 font-medium">已启用 → 点击关闭</span>
                      ) : (
                        <span className="text-gray-500">未启用 → 点击开启</span>
                      )}
                    </span>
                  </label>
                </div>

                <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                  <div>
                    <Label text={`Temperature（${draft.temperature}）`} hint="越高越发散，越低越稳定" />
                    <Slider
                      min={0}
                      max={2}
                      step={0.1}
                      value={draft.temperature}
                      onChange={(v) => setDraft({ ...draft, temperature: Number(v) })}
                    />
                  </div>
                  <div>
                    <Label text="Timeout（秒）" hint="单次请求最长等待时间" />
                    <InputNumber
                      value={draft.timeout_seconds}
                      min={1}
                      onChange={(v) => setDraft({ ...draft, timeout_seconds: Number(v || 0) })}
                      style={{ width: '100%' }}
                    />
                  </div>
                  <div>
                    <Label text="Max Tokens" hint="单次响应最大 token 数" />
                    <InputNumber
                      value={draft.max_tokens}
                      min={1}
                      onChange={(v) => setDraft({ ...draft, max_tokens: Number(v || 0) })}
                      style={{ width: '100%' }}
                    />
                  </div>
                </div>

                <div>
                  <Label
                    text="主要用途（Primary For）"
                    hint="仅用于服务内部路由：chat / embedding / summarizer / reasoning"
                  />
                  <Checkbox.Group
                    value={draft.primary_for}
                    onChange={(vals) => setDraft({ ...draft, primary_for: vals as string[] })}
                    options={ALL_PURPOSES.map((p) => ({ label: p, value: p }))}
                  />
                </div>
              </div>
          </Section>

          {/* 底部吸附操作栏：只保留保存 */}
          <div className="fixed bottom-0 left-64 right-0 bg-white/90 backdrop-blur border-t border-gray-200 px-8 py-3 flex items-center justify-between z-40">
            <div className="text-sm text-gray-600 flex items-center gap-2 flex-wrap">
              {draft.enabled ? <Tag color="green">已启用</Tag> : <Tag>未启用</Tag>}
              {draft.has_secret && <Tag color="blue">密钥就绪</Tag>}
              {validating && <Tag color="processing">连通性检测中…</Tag>}
              {!validating && validated && <Tag color="success">连接正常</Tag>}
              {!validating && !validated && draft.default_model && error && (
                <span className="text-xs text-amber-700">连接异常，请检查上方提示</span>
              )}
            </div>
            <Button type="primary" size="large" onClick={saveConfig} loading={savingConfig} icon={<Check className="w-4 h-4" />}>
              保存配置
            </Button>
          </div>
        </>
      )}
    </div>
  )
}

// ---------- 子组件 ----------

function Section({
  step,
  title,
  subtitle,
  children,
}: {
  step?: number
  title: string
  subtitle?: React.ReactNode
  children: React.ReactNode
}) {
  return (
    <section className="bg-white rounded-xl shadow-sm border border-gray-100">
      <header className="flex items-start justify-between gap-4 px-6 py-4">
        <div className="flex items-start gap-3">
          {typeof step === 'number' && (
            <span className="w-7 h-7 rounded-full bg-blue-600 text-white text-sm font-semibold flex items-center justify-center shrink-0">
              {step}
            </span>
          )}
          <div>
            <h2 className="text-base font-semibold text-gray-900">{title}</h2>
            {subtitle && <div className="text-xs text-gray-500 mt-0.5">{subtitle}</div>}
          </div>
        </div>
      </header>
      <div className="px-6 pb-6 pt-2">{children}</div>
    </section>
  )
}

function Label({
  text,
  hint,
  badge,
}: {
  text: string
  hint?: string
  badge?: React.ReactNode
}) {
  return (
    <div className="flex items-center justify-between mb-1.5">
      <div className="flex items-center gap-1.5">
        <span className="text-sm font-medium text-gray-700">{text}</span>
        {hint && (
          <Tooltip title={hint}>
            <AlertCircle className="w-3.5 h-3.5 text-gray-400 cursor-help" />
          </Tooltip>
        )}
      </div>
      {badge}
    </div>
  )
}

function StatusBanner({
  tone,
  title,
  description,
  notice,
}: {
  tone: 'success' | 'warning' | 'error' | 'info'
  title: string
  description: string
  notice?: string
}) {
  return (
    <Alert
      type={tone}
      showIcon
      message={<span className="font-semibold">{title}</span>}
      description={
        <div className="space-y-1">
          <div>{description}</div>
          {notice && <div className="text-xs text-green-700">{notice}</div>}
        </div>
      }
    />
  )
}
